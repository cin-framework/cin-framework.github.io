# Function to create a new folder
function New-CustomFolder {
    param (
        [Parameter(Mandatory=$true)]
        [string]$Path,
        [Parameter(Mandatory=$true)]
        [string]$FolderName
    )
    $FullFolderPath = Join-Path -Path $Path -ChildPath $FolderName
    if (-not (Test-Path $FullFolderPath)) {
        New-Item -Path $FullFolderPath -ItemType Directory
        Write-Host "Folder '$FolderName' created at '$Path'."
    } else {
        Write-Host "Folder '$FolderName' already exists at '$Path'."
    }
}

# Example usage of the function
New-CustomFolder -Path "C:\Temp" -FolderName "MyNewProject"
